struct Record *add_record(struct Record *database, int *record_count, int *max_capacity) {
    if (*record_count == *max_capacity) {
        *max_capacity *= 2;

        // Allocate to a temp pointer first to prevent memory leak if realloc fails
        struct Record *temp = realloc(database, *max_capacity * sizeof(struct Record));
        if (temp == NULL) {
            printf("Fatal Error: Out of memory!\n");
            free(database);
            exit(1);
        }
        database = temp;
    }

    printf("Enter id: ");
    scanf("%d", &database[*record_count].id);

    // Flush stdin buffer to prevent fgets from skipping on the next read
    char ch;
    while ((ch = getchar()) != '\n')
        ;

    printf("Enter name: ");
    char *name = database[*record_count].name;
    fgets(name, 21, stdin);
    name[strcspn(name, "\n")] = '\0';

    printf("Enter age: ");
    scanf("%d", &database[*record_count].age);
    while ((ch = getchar()) != '\n')
        ;

    (*record_count)++;

    return database;
}
